# Common git comands

1. git init
2. git add .
3. git reset .
4. git commit
5. git checkout -- . (to recover the last commit)
6. git log
7. git commit --amend
8. git checkout -b new-branch
9. git checkout main
10. git branch -d rama-heroes
11. git remote add origin url repo + .git
12. git push
13. git commit -am
